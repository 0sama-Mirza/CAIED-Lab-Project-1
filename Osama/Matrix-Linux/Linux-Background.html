My Script To Change Wallpaper on Gnome


#!/bin/bash
wallpaperdir='/home/baymax/Pictures/wallpaper'
while :
do
    pic=$(ls $wallpaperdir/* | shuf -n1)
    gsettings set org.gnome.desktop.background picture-uri-dark "file://$pic"
    pic=$(ls $wallpaperdir/* | shuf -n1)
    gsettings set org.gnome.desktop.background picture-uri "file://$pic"
    sleep 60
done
